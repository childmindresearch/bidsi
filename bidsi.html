131    def _merge(
132        self,
133        path: Path,
134        resource: T,
135        merge_strategy: MergeStrategy,
136        merge_op: Callable[[Path, T], T],
137        write_op: Callable[[Path, T], None],
138    ) -> None:
139        """Merge file with BIDS structure respecting Merge config."""
140        LOG.debug(f"Merging resource to {path}")
141        # Set path and resource according to merge_strategy.
142        # Raise error if path exists and merge_strategy is NO_MERGE.
143        # Return early to avoid writing.
144        match merge_strategy:
145            case MergeStrategy.NO_MERGE:
146                if path.exists():
147                    raise ValueError(f"File {path} already exists.")
148            case MergeStrategy.OVERWRITE:
149                if path.exists():
150                    LOG.debug(f"Overwriting file at {path}")
151            case MergeStrategy.KEEP:
152                if path.exists():
153                    LOG.debug(f"Keeping existing file at {path}")
154                    return
155            case MergeStrategy.RENAME_SEQUENTIAL:
156                # Set Path to new file with run-label increment
157                pass
158            case MergeStrategy.MERGE:
159                if path.exists():
160                    resource = merge_op(path, resource)
161        write_op(path, resource)

163    def _merge_entity(self, entity: BidsEntity, dir: Path) -> None:
164        """Write file respecting MergeStrategy."""
165        expected_filepath = dir / self._config.entity.entity_name(entity)
166        LOG.debug(f"Merging entity to {expected_filepath}")
167        self._ensure_directory_path(expected_filepath)
168        if entity.file_path is not None:

182                self._config.merge.entity,
183                self._merge_tabular,
184                self._write_tabular,
185            )
186        else:
187            raise ValueError(f"Unknown entity type for {entity}")
188        if entity.metadata:

202    def _merge_folder(
203        self, path: Path, merge_strategy: MergeStrategy
204    ) -> Tuple[Path, bool]:
205        """Merge folder respecting MergeStrategy."""
206        LOG.debug(f"Merging folder to {path} with strategy {merge_strategy}")
207        match merge_strategy:
208            case MergeStrategy.NO_MERGE:
209                if path.exists():
210                    if not path.is_dir():
211                        raise ValueError(f"Path {path} is not a directory.")
212                    if len(list(path.iterdir())) > 0:
213                        raise ValueError(
214                            f"Folder {path} is not empty "
215                            "and merge_strategy is NO_MERGE."
216                        )
217                # Path does not exist or is empty. Create directory for writing.
218                self._ensure_directory_path(path, is_dir=True)
219                return path, True
220            case MergeStrategy.OVERWRITE:
221                # Remove and recreate directory.
222                shutil.rmtree(path, ignore_errors=True)
223                self._ensure_directory_path(path, is_dir=True)
224                return path, True
225            case MergeStrategy.KEEP:
226                self._ensure_directory_path(path, is_dir=True)
227                return path, False
228            case MergeStrategy.RENAME_SEQUENTIAL:
229                # TODO: Return folder with run-label increment
230                raise NotImplementedError(
231                    "RENAME_SEQUENTIAL merge for directories not implemented."
232                )
233            case MergeStrategy.MERGE:
234                self._ensure_directory_path(path, is_dir=True)
235                return path, True
